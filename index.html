Extensions → Apps Script // 3. วางโค้ดนี้ทั้งหมดแทนที่โค้ดเดิม // 4. บันทึก → Deploy → New deployment // - Type: Web app // - Execute as: Me // - Who has access: Anyone // 5. คัดลอก Web app URL ไปใส่ในไฟล์ HTML // ═══════════════════════════════════════════════════════════════ const SHEET_NAMES = { depts: 'Departments', projects: 'Projects', expenses: 'Expenses', users: 'Users', meta: 'Meta' }; // ── CORS / Entry point ────────────────────────────────────────── function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { const params = e.parameter || {}; const body = e.postData ? JSON.parse(e.postData.contents || '{}') : {}; const action = params.action || body.action; let result; try { switch (action) { case 'login': result = actionLogin(body); break; case 'getData': result = actionGetData(body); break; case 'saveData': result = actionSaveData(body); break; case 'getYears': result = actionGetYears(); break; case 'setup': result = actionSetup(); break; default: result = { ok: false, error: 'Unknown action: ' + action }; } } catch (err) { result = { ok: false, error: err.toString() }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } // ── Sheet helpers ─────────────────────────────────────────────── function getSheet(name) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sh = ss.getSheetByName(name); if (!sh) sh = ss.insertSheet(name); return sh; } function sheetToArray(sh) { const data = sh.getDataRange().getValues(); if (data.length < 2) return []; const headers = data[0]; return data.slice(1).map(row => { const obj = {}; headers.forEach((h, i) => { obj[h] = row[i]; }); return obj; }); } function clearAndWrite(sh, headers, rows) { sh.clearContents(); if (rows.length === 0) { sh.getRange(1, 1, 1, headers.length).setValues([headers]); return; } const data = [headers, ...rows.map(r => headers.map(h => { const v = r[h]; // Serialize objects/arrays as JSON strings if (typeof v === 'object' && v !== null) return JSON.stringify(v); return v === undefined ? '' : v; }))]; sh.getRange(1, 1, data.length, headers.length).setValues(data); } // ── Setup: สร้าง sheets และ admin user เริ่มต้น ──────────────── function actionSetup() { // Departments sheet const dSh = getSheet(SHEET_NAMES.depts); if (dSh.getLastRow() < 2) { clearAndWrite(dSh, ['id','name','color','pfx'], [ {id:1, name:'ฝ่ายการตลาด', color:'#D85A30', pfx:'MKT'}, {id:2, name:'ฝ่ายไอที', color:'#534AB7', pfx:'IT'}, {id:3, name:'ฝ่าย HR', color:'#1D9E75', pfx:'HR'}, ]); } // Projects sheet const pSh = getSheet(SHEET_NAMES.projects); if (pSh.getLastRow() < 2) { const year = new Date().getFullYear(); clearAndWrite(pSh, ['id','year','dId','code','name','bud','totalBud','monthlyBuds','month','desc'], [ {id:101,year,dId:1,code:'MKT-001',name:'แคมเปญดิจิทัล', bud:500000, totalBud:500000, monthlyBuds:'{}', month:`${year}-01`, desc:''}, {id:102,year,dId:1,code:'MKT-002',name:'งานแสดงสินค้า', bud:200000, totalBud:200000, monthlyBuds:'{}', month:`${year}-01`, desc:''}, {id:103,year,dId:2,code:'IT-001', name:'อัปเกรดระบบ ERP', bud:1200000,totalBud:1200000,monthlyBuds:'{}', month:`${year}-01`, desc:''}, {id:104,year,dId:2,code:'IT-002', name:'Cybersecurity Audit',bud:350000, totalBud:350000, monthlyBuds:'{}', month:`${year}-01`, desc:''}, {id:105,year,dId:3,code:'HR-001', name:'โปรแกรมอบรมพนักงาน',bud:300000, totalBud:300000, monthlyBuds:'{}', month:`${year}-01`, desc:''}, ]); } // Expenses sheet const eSh = getSheet(SHEET_NAMES.expenses); if (eSh.getLastRow() < 1) { clearAndWrite(eSh, ['id','year','dId','pId','date','type','desc','amt'], []); } // Users sheet const uSh = getSheet(SHEET_NAMES.users); if (uSh.getLastRow() < 2) { clearAndWrite(uSh, ['username','pass','role','deptId','label'], [ {username:'admin', pass:'admin1234', role:'admin', deptId:'', label:'ผู้ดูแลระบบ'}, {username:'viewer', pass:'view1234', role:'viewer', deptId:'', label:'ผู้ดูข้อมูล (ทุกฝ่าย)'}, {username:'mkt', pass:'mkt1234', role:'viewer', deptId:1, label:'ผู้ดูข้อมูล (การตลาด)'}, {username:'it', pass:'it1234', role:'viewer', deptId:2, label:'ผู้ดูข้อมูล (ไอที)'}, {username:'hr', pass:'hr1234', role:'viewer', deptId:3, label:'ผู้ดูข้อมูล (HR)'}, ]); } return { ok: true, message: 'Setup complete' }; } // ── Login ─────────────────────────────────────────────────────── function actionLogin(body) { const { username, pass } = body; const users = sheetToArray(getSheet(SHEET_NAMES.users)); const u = users.find(x => x.username === username && x.pass === pass); if (!u) return { ok: false, error: 'ชื่อผู้ใช้หรือรหัสผ่านไม่ถูกต้อง' }; return { ok: true, user: { username: u.username, role: u.role, deptId: u.deptId ? +u.deptId : null, label: u.label } }; } // ── Get data for a year ───────────────────────────────────────── function actionGetData(body) { const year = +body.year || new Date().getFullYear(); const depts = sheetToArray(getSheet(SHEET_NAMES.depts)).map(d => ({ id: +d.id, name: d.name, color: d.color, pfx: d.pfx })); const allProjects = sheetToArray(getSheet(SHEET_NAMES.projects)); const projects = allProjects .filter(p => +p.year === year) .map(p => ({ id: +p.id, year: +p.year, dId: +p.dId, code: p.code, name: p.name, bud: +p.bud, totalBud: +p.totalBud, monthlyBuds: parseJSON(p.monthlyBuds, {}), month: p.month, desc: p.desc })); const allExpenses = sheetToArray(getSheet(SHEET_NAMES.expenses)); const expenses = allExpenses .filter(e => +e.year === year) .map(e => ({ id: +e.id, year: +e.year, dId: +e.dId, pId: +e.pId, date: e.date, type: e.type, desc: e.desc, amt: +e.amt })); const users = sheetToArray(getSheet(SHEET_NAMES.users)).map(u => ({ username: u.username, role: u.role, deptId: u.deptId ? +u.deptId : null, label: u.label // ไม่ส่ง pass กลับ })); return { ok: true, depts, projects, expenses, users }; } // ── Get all years that have project data ──────────────────────── function actionGetYears() { const projects = sheetToArray(getSheet(SHEET_NAMES.projects)); const years = [...new Set(projects.map(p => +p.year).filter(Boolean))]; if (!years.length) years.push(new Date().getFullYear()); years.sort((a, b) => b - a); return { ok: true, years }; } // ── Save all data for a year ──────────────────────────────────── function actionSaveData(body) { const year = +body.year; const depts = body.depts || []; const projects= body.projects|| []; const expenses= body.expenses|| []; const users = body.users; // optional if (!year) return { ok: false, error: 'Missing year' }; // ── Departments (shared across years) ── const dSh = getSheet(SHEET_NAMES.depts); clearAndWrite(dSh, ['id','name','color','pfx'], depts); // ── Projects: replace only records for this year ── const pSh = getSheet(SHEET_NAMES.projects); const allP = sheetToArray(pSh).filter(p => +p.year !== year); // keep other years const newP = projects.map(p => ({ id: p.id, year, dId: p.dId, code: p.code, name: p.name, bud: p.bud, totalBud: p.totalBud, monthlyBuds: JSON.stringify(p.monthlyBuds || {}), month: p.month, desc: p.desc || '' })); clearAndWrite(pSh, ['id','year','dId','code','name','bud','totalBud','monthlyBuds','month','desc'], [...allP, ...newP]); // ── Expenses: replace only records for this year ── const eSh = getSheet(SHEET_NAMES.expenses); const allE = sheetToArray(eSh).filter(e => +e.year !== year); const newE = expenses.map(e => ({ id: e.id, year, dId: e.dId, pId: e.pId, date: e.date, type: e.type, desc: e.desc || '', amt: e.amt })); clearAndWrite(eSh, ['id','year','dId','pId','date','type','desc','amt'], [...allE, ...newE]); // ── Users (optional update) ── if (users) { const uSh = getSheet(SHEET_NAMES.users); const existingUsers = sheetToArray(uSh); // Merge: keep passwords from existing, update role/deptId/label const merged = users.map(u => { const ex = existingUsers.find(x => x.username === u.username); return { username: u.username, pass: ex ? ex.pass : (u.pass || 'changeme'), role: u.role, deptId: u.deptId || '', label: u.label }; }); // Add any users not in new list (preserve) existingUsers.forEach(ex => { if (!merged.find(m => m.username === ex.username)) { merged.push(ex); } }); clearAndWrite(uSh, ['username','pass','role','deptId','label'], merged); } return { ok: true }; } // ── Helper ────────────────────────────────────────────────────── function parseJSON(str, fallback) { try { return JSON.parse(str); } catch(e) { return fallback; } }
